<script>

  const grid = document.getElementById('grid');

  const revealBox = document.getElementById('revealBox');

  const resetBtn = document.getElementById('resetBtn');

  const triesEl = document.getElementById('tries');

  const bar = document.getElementById('bar');


  const CLOUDS = 12;

  let target = 0;

  let opened = 0;

  let gameDone = false;


  const cloudEmojis = ["☁️","🌤️","☁️","🌥️","☁️","🌦️","☁️","🌤️","☁️","🌥️","☁️","🌤️"];


  function build(){

    grid.innerHTML = "";

    revealBox.style.display = "none";

    opened = 0;

    gameDone = false;

    updateUI();

    target = Math.floor(Math.random() * CLOUDS);


    for(let i=0;i<CLOUDS;i++){

      const tile = document.createElement('div');

      tile.className = "cloud";

      tile.setAttribute("role","button");

      tile.setAttribute("aria-label","Cloud");


      const sparkle = document.createElement('div');

      sparkle.className = "sparkle";


      const emoji = document.createElement('div');

      emoji.className = "emoji";

      emoji.textContent = cloudEmojis[i % cloudEmojis.length];


      const tag = document.createElement('div');

      tag.className = "tag";

      tag.textContent = "TAP ME!";


      tile.appendChild(sparkle);

      tile.appendChild(emoji);

      tile.appendChild(tag);


      tile.addEventListener('click', () => openCloud(tile, i), { once:true });


      grid.appendChild(tile);

    }

  }


  function updateUI(){

    triesEl.textContent = `Clouds opened: ${opened}`;

    const pct = Math.min(100, (opened / CLOUDS) * 100);

    bar.style.width = pct + "%";

  }


  function openCloud(tile, idx){

    if(gameDone) return;


    opened++;

    updateUI();


    // If it's the winning cloud, reveal instantly (no pop animation)

    if(idx === target){

      gameDone = true;

      tile.style.cursor = "default";

      tile.querySelector(".emoji").textContent = "⭐";

      tile.querySelector(".tag").textContent = "YOU FOUND IT!";

      setTimeout(() => {

        revealBox.style.display = "block";

        launchConfetti();

        // make the other clouds gently fade

        [...document.querySelectorAll('.cloud')].forEach((c, j) => {

          if(j !== idx) c.style.opacity = "0.55";

        });

      }, 250);

      return;

    }


    // Not the target: cute pop away + tiny delay

    tile.classList.add("opened");

  }


  // Confetti (simple, no libraries)

  const canvas = document.getElementById("confetti");

  const ctx = canvas.getContext("2d");

  let raf = null;


  function resizeCanvas(){

    canvas.width = window.innerWidth;

    canvas.height = window.innerHeight;

  }

  window.addEventListener("resize", resizeCanvas);


  function launchConfetti(){

    resizeCanvas();

    canvas.style.display = "block";


    const colors = ["#ff4d8d","#ff85b3","#ffd1e6","#c084fc","#5eead4","#fbbf24"];

    const pieces = Array.from({length: 170}, () => ({

      x: Math.random()*canvas.width,

      y: -20 - Math.random()*canvas.height*0.25,

      w: 6 + Math.random()*8,

      h: 4 + Math.random()*7,

      c: colors[Math.floor(Math.random()*colors.length)],

      vx: -2.2 + Math.random()*4.4,

      vy: 2.0 + Math.random()*5.0,

      rot: Math.random()*Math.PI,

      vr: -0.12 + Math.random()*0.24

    }));


    const start = performance.now();

    const DURATION = 2600;


    if(raf) cancelAnimationFrame(raf);


    function frame(t){

      const elapsed = t - start;

      ctx.clearRect(0,0,canvas.width,canvas.height);


      for(const p of pieces){

        p.x += p.vx;

        p.y += p.vy;

        p.rot += p.vr;

        p.vy += 0.03; // gravity


        ctx.save();

        ctx.translate(p.x, p.y);

        ctx.rotate(p.rot);

        ctx.fillStyle = p.c;

        ctx.fillRect(-p.w/2, -p.h/2, p.w, p.h);

        ctx.restore();

      }


      if(elapsed < DURATION){

        raf = requestAnimationFrame(frame);

      }else{

        canvas.style.display = "none";

      }

    }

    raf = requestAnimationFrame(frame);

  }


  resetBtn.addEventListener('click', build);

  build();
